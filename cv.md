## Code example

```js
// https://www.codewars.com/kata/62eb800ba29959001c07dfee

function brightest(colors) {
	const result = colors.map(i => {
		const rgb = i.split('#')[1].match(/.{2}/g).map(i => (parseInt(i, 16) / 255))
		return [i, Math.max(...rgb)]
	}).sort((a, b) => b[1] - a[1])[0][0]
	return result
}
```
